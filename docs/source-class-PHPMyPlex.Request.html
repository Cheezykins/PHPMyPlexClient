 39:         'X-Plex-Platform' => PHP_OS,
 40:         'X-Plex-Platform-Version' => false,
 41:         'X-Plex-Provides' => 'controller',
 42:         'X-Plex-Product' => 'PHPMyPlex',
 43:         'X-Plex-Version' => false,
 44:         'X-Plex-Device' => false,
 45:         'X-Plex-Client-Identifier' => false,
 46:     ];
 47:     private $template;
 48:     private $endPoint;

 50:     /**
 51:      * Creates an HTTP Request manager with additional header support for Plex.
 52:      * Endpoint is the URL to request from, with optional proxy parameter.
 53:      * expectEmptyResponse when set to true does not try and parse the result as XML.
 54:      * 
 55:      * @param string $endPoint
 56:      * @param bool|PHPMyPlex\Proxy $proxy
 57:      */
 58:     public function __construct($endPoint, $proxy = false, $expectEmptyResponse = false)
 59:     {
 60:         $this->endPoint = $endPoint;
 61:         $this->template = HRequest::init();
 62:         if ($expectEmptyResponse) {
 63:             $this->template->withoutAutoParse();
 64:         } else {
 65:             $this->template->expectsXml();
 66:         }
 67:         if ($proxy) {
 68:             $this->template->useProxy($proxy->host, $proxy->port, $proxy->authType);
 69:         }
 70:         foreach ($this->headers as $header => $value) {
 71:             if ($value) {
 72:                 $this->setHeader($header, $value);
 73:             }
 74:         }
 75:     }
 76: 
 77:     /**
 78:      * Helper method allows headers to be set as parameters.
 79:      * 
 80:      * @param string $name
 81:      * @param string $value
 82:      */
 83:     public function __set($name, $value)
 84:     {
 85:         $this->setHeader($name, $value);
 86:     }
 87: 
 88:     /**
 89:      * Sets a header in the HTTP request.
 90:      * Headers in camelCase are transposed to X-Plex-Camel-Case
 91:      * 
 92:      * @param string $header
 93:      * @param string $value
 94:      */
 95:     public function setHeader($header, $value)
 96:     {
 97:         if (\substr($header, 0, 7) != 'X-Plex-') {
 98:             $header = 'X-Plex-' . \ucfirst(\preg_replace('/([A-Z])/', '-$0', $header));
 99:         }
100:         $this->headers[$header] = $value;
101:         $this->template->addHeader($header, $value);
102:     }
103: 
104:     /**

107:      * 
108:      * @param string $userName
109:      * @param string $password

118:      * @param string $name
119:      * @return string

121:     public function __get($name)

128:     /**
129:      * Initiates the HTTP request using the provided method (GET, POST, PUT, DELETE)
130:      * and returns the response XML as an Httpful\Request
131:      * 
132:      * @param string $method
133:      * @return Httpful\Request
134:      * @throws Exceptions\MyPlexDataException
135:      */
136:     public function send($method)
137:     {
138:         HRequest::ini($this->template);
139:         try {
140:             $response = HRequest::{$method}($this->endPoint)->send();
141:         } catch (Httpful\Exception\ConnectionErrorException $e) {
142:             throw new Exceptions\MyPlexDataException('Unable to connect to endPoint: ' . $e->getMessage(), 0, $e);
143:         } catch (\Exception $e) {
144:             throw new Exceptions\MyPlexDataException('Error in response from server: ' . $e->getMessage(), 0, $e);
145:         }
146:         $this->errorCheck($response);
147:         return $response;
148:     }
149: 
150:     /**
151:      * Checks the response from the HTTP request for a variety of errors and issues
152:      * and raises appropriate exceptions if detected.
153:      * 
154:      * @param Httpful\Request $response
155:      * @throws Exceptions\MyPlexAuthenticationException
156:      * @throws Exceptions\MyPlexDataException
157:      */
158:     private function errorCheck($response)
159:     {
160:         if ($response->hasErrors()) {
161:             if ($response->code == 401) {
162:                 throw new Exceptions\MyPlexAuthenticationException((string) $response->body->error);
163:             }
164: 
165:             $error = 'Error code ' . $response->code . ' recieved from server';
166:             if ($response->hasBody() && isset($response->body->error)) {
167:                 $error .= ': ' . (string) $response->body->error;
168:             }
169:             throw new Exceptions\MyPlexDataException($error);
170:         }
171:     }
