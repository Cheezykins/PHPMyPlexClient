33: class MediaContainerCollection extends \ArrayObject

36:     /**
37:      * Iterates through the collection and loads all sub-items.
38:      * Returns itself for chaining.
39:      * @return \PHPMyPlex\Containers\MediaContainerCollection
40:      */
41:     public function loadAll()
42:     {
43:         foreach ($this as $obj) {
44:             $obj->load();

46:         return $this;
47:     }

50:      * Gets all details for containers within the collection.
51:      * @return array

54:     {

57:             $details[] = $obj->getDetails();
58:         }
59:         return $details;
60:     }

66:      * @return \PHPMyPlex\Containers\MediaContainerCollection
67:      */
68:     public function children()
69:     {
